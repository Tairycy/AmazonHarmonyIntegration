_bt_avrcp_mgr_t Structure
C
struct _bt_avrcp_mgr_t {
  bt_byte state;
  bt_byte flags;
  bt_ulong company_id;
  bt_uint supported_events;
  bt_ulong song_length;
  bt_ulong song_position;
  bt_byte play_status;
  bt_byte volume;
  bt_byte battery_status;
  bt_byte system_status;
  bt_av_element_id_t current_track_id;
  bt_uint uid_counter;
  bt_avrcp_channel_t* channels;
  bt_avrcp_mgr_callback_fp callback;
  void* callback_param;
};
Description

brief AVRCP manager. ingroup avrcp 

details A structure that glues all pieces together. There is only one instance of this structure allocated by dotstack. A pointer to the instance can be get with c bt_avrcp_get_mgr().
